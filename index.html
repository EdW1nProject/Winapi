// Contoh menghubungkan proyek e-commerce Anda ke WINYUK Upload API
const uploadFoto = async (base64String, fileName) => {
    try {
        const response = await fetch('https://your-vercel-domain.com/api/upload', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                image: base64String,
                filename: fileName
            })
        });

        const data = await response.json();

        if (data.success) {
            console.log("Berhasil Upload! URL URL:", data.url);
        } else {
            console.error("Gagal:", data.message);
        }
    } catch (error) {
        console.error("Koneksi API Error:", error);
    }
};
