const axios = require('axios');

// WINYUK API - TikTok Downloader Example
const API_KEY = 'YOUR_API_KEY';
const TIKTOK_URL = 'https://vt.tiktok.com/ZSxxxxxxx/';

async function downloadTikTok(url) {
    try {
        const res = await axios.get(`https://api.winyuk.com/v1/dl/tiktok?url=${url}`, {
            headers: { 'Authorization': `Bearer ${API_KEY}` }
        });
        
        console.log('Title:', res.data.title);
        console.log('Video (No WM):', res.data.video_nowm);
        if(res.data.is_slide) {
             console.log('Photos:', res.data.images); // Array of images
        }
    } catch (err) {
        console.error('Error:', err.response?.data || err.message);
    }
}

downloadTikTok(TIKTOK_URL);
